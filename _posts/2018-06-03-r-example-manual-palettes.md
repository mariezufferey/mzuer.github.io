Few suggestions for manual palettes

```
# 4 color palette + grey
values <- c("#CCCCCC","#9966FF","#33FFFF","#669900","#FF6666")

# 4 color palette
values <- c("#9966FF","#33FFFF","#669900","#FF6666")

# 3 color palette + grey
values <- c("#CCCCCC","#FFFF00","#FF0000","#003300")

# 5 color palette (flashy)
values <- c("#66FF00","#FFFF00","#FF0000","#FF0099","#9900FF")

# 6 color palette
values <- c("#33CC00","#33CCFF","#FF9933","#CC33FF","#FFFF33","#FF0000"))

# 7 color palette
values <- c("#666666","#FFFF00","#FF0000","#FF9933","#FF0099","#9900FF","#3399FF")
```
